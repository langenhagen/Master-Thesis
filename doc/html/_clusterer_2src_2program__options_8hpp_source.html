135  bool init_program_options( int argc, const char* argv[], program_options& p) {
136  using namespace boost::program_options;
137  using namespace std;
138  bool ret(true);
139 
140  // declare the options ************************************************
141 
142  // 1. group of options that will be allowed only on command line
143  options_description cmd_desc("Command line options options");
144  cmd_desc.add_options()
145  ("help", "produce help message")
146  ("config,c", value<string>(&p.config_file)->default_value("config.cfg"), "name of the config file")
147  ;
148  // 2. option group description on the config file
149  options_description config_desc("Configuration file parameters");
150  config_desc.add_options()
151  ("log_file", value<string>(&p.log_file)->default_value("log.log"), "name of the log file")
152  ("features_file", value<string>(&p.features_file)->default_value("features.txt"), "a file that stores the feature vectors")
153  ("images_file", value<string>(&p.images_file), "a file that stores the paths of the images")
154  ("clusterer_type", value<string>(&p.cd.type_string), clusterer_types_string().c_str())
155  ("clusterer_tweak_vector", value<string>(&p.cd.tweak_vector_string)->default_value(""), "real-numeric tweaks for the feature extractor separated by spaces \" \".")
156  ("output_directory", value<string>(&p.output_directory)->default_value("out"), "the output directory for output-files")
157  ("membership_probabilities_file", value<string>(&p.membership_probabilities_file)->default_value("membership_probabilities.txt"), "Stores the probabilities of each feature to belong to each cluster")
158  ("membership_mappings_file", value<string>(&p.membership_mappings_file)->default_value("membership_mappings.txt"), "Stores the index of the cluster with the highest membership-probability for each feature")
159  ("cluster_means_file", value<string>(&p.cluster_means_file)->default_value("cluster_means.txt"), "Stores the centers of each cluster in this file")
160  ("result_file_prefix", value<string>(&p.result_file_prefix)->default_value("class_"), "the prefix of the files that contain the classification results")
161  ("result_filenames_file", value<string>(&p.result_fnames_file)->default_value("class_files"), "the file that stores the paths to all result files")
162  ("symlink_results", value<bool>(&p.symlink_results)->default_value(0), "whether or not to symlink the images into folders named after their classes")
163  ("symlink_saliency_maps", value<bool>(&p.symlink_saliency_maps)->default_value(0), "whether or not to symlink the saliency maps that are eventually generated by the FeatureGenerator")
164  ("saliency_maps_file", value<string>(&p.saliency_maps_file)->default_value("saliency_maps.txt"), "the file that stores the paths to all result saliency maps.")
165  ;
166  // END option declarations ********************************************
167 
168  // *** parse the command line and the config files ***
169  variables_map vm;
170 
171  // 1. parse command line ******************************************
172  store(parse_command_line(argc, argv, cmd_desc), vm);
173  boost::program_options::notify(vm);
174 
175  if( vm.count("help")) {
176  // special 'help' case
177  print_help_message( 2, &cmd_desc, &config_desc);
178  return false;
179  }
180  // 2. parse config file *******************************************
181  ifstream in_file(p.config_file.c_str());
182  if (!in_file) {
183  std::cerr << "Can not open config file: " << p.config_file << "\n"
184  "Did you create a config file or do you want to specify another config file?\n";
185  ret = false;
186  } else {
187  // *** file access works ***
188  parsed_options po (parse_config_file(in_file, config_desc, true));
189  store( po, vm);
190  boost::program_options::notify(vm);
191  }
192  in_file.close();
193  return ret;
194  }
