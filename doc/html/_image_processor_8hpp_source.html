133  bool process_image( const boost::filesystem::path& image_path, const Mat3b& image) {
134  bool ret(true);
135  Mat1b saliency_map, saliency_mask;
136  vector<Contour> contours;
137  Vec1r features;
138 
139  try {
140  saliency_map = _saliency_detector->saliency(image);
141  } catch( std::exception& e) {
142  saliency_map = Mat1b::zeros(image.rows, image.cols);
143  LOG(exception) << "Failed to extract saliency map!\n" <<
144  e.what();
145  }
146 
147  saliency_mask = generate_saliency_mask( image, saliency_map, contours);
148 
149  if(contours.size() == 0) {
150  // *** no salient region found ***
151  LOG(notify) << "No salient region found in \"" << image_path.string() << "\".";
152  ret = handle_garbage_file( image_path);
153  } else {
154  // *** salient regions found ***
155  return_error_code::return_error_code ec = return_error_code::UNSPECIFIED_ERROR;
156  try {
157  ec = _feature_extractor->extract(image, saliency_map, saliency_mask, contours, features);
158  } catch( std::exception& e) {
159  LOG(exception) << "Failed to extract feature vector!\n" <<
160  e.what();
161  }
162 
163 
164  // *** error with extracted features ***
165  switch( ec) {
166  case return_error_code::SUCCESS:
167  // *** features successfully extracted ***
168 
169  _features_fstream << features << "\n";
170  ANXIOUS_FLUSH(_features_fstream)
171 
172  if( _features_fstream.bad()) {
173  LOG(error) << "Failed writing feature vector to file \"" << params.features_file << "\"!";
174  ret = false;
175  }
176  _processed_images_fstream << image_path.string() << "\n";
177  ANXIOUS_FLUSH(_processed_images_fstream)
178 
179  if( _processed_images_fstream.bad()) {
180  LOG(error) << "Failed writing to file \"" << params.processed_images_file << "\"!";
181  ret = false;
182  }
183 
184  // store intermediate results
185  imwrite_if( params.save_saliency_maps, image_path, saliency_map, "_saliency.png", _saliency_maps_fstream, params.saliency_maps_file, contours.size()>0);
186  imwrite_if( params.save_saliency_masks, image_path, saliency_mask, "_saliency_mask.png", _saliency_masks_fstream, params.saliency_masks_file);
187 
188  stats.n_processed_images++;
189  stats.n_processed_images_in_current_session++;
190  break;
191 
192  case return_error_code::INFINITE_NUMBER_ERROR:
193  LOG(error) << "Feature extraction error (infinite number) while processing \"" << image_path.string() << "\". Image is considered garbage.";
194  ret = handle_garbage_file( image_path);
195  break;
196  case return_error_code::NAN_ERROR:
197  LOG(error) << "Feature extraction error (NAN) while processing \"" << image_path.string() << "\". Image is considered garbage.";
198  ret = handle_garbage_file( image_path);
199  break;
200  case return_error_code::UNSPECIFIED_ERROR:
201  default:
202  LOG(error) << "Feature extraction error (unknown error) while processing \"" << image_path.string() << "\". Image is considered garbage.";
203  ret = handle_garbage_file( image_path);
204  }
205  }
206  return ret;
207  }

292  Mat1b generate_saliency_mask( const Mat3b& image, const Mat1b& saliency_map, vector<Contour>& o_contours) {
293  using namespace cv;
294  Mat1b ret;
295 
296  if( params.use_grabcut) {
297  // Grabcut
298  ret = grabcut( image, saliency_map);
299 
300  } else {
301  // classic treshold
302  GaussianBlur( saliency_map, ret, params.blur_kernel_size, 0, 0);
303  threshold( ret, ret, 255*params.threshold, 255, CV_THRESH_BINARY);
304  //threshold( ret, ret, 255*params.threshold, 255, CV_THRESH_OTSU);
305  }
306 
307  // *** smoothed top-level contours left ***
308  o_contours = generate_contours( ret);
309  ret = Mat::zeros(ret.rows, ret.cols, CV_8UC1);
310  drawContours(ret, o_contours, -1, cv::Scalar(255), CV_FILLED);
311  return ret;
312  }

321  Mat1b grabcut( const Mat3b& image, const Mat1b& saliency_map) {
322  using namespace cv;
323  Mat1b ret( saliency_map.size(), cv::GC_BGD);
324 
325  for( int r=0; r<ret.rows; ++r) {
326  for( int c=0; c<ret.cols; ++c) {
327  if( saliency_map(r,c) > 255 * params.grabcut_foreground_probability)
328  ret(r,c) = cv::GC_PR_FGD;
329  }}
330 
331  // GrabCut segmentation
332  cv::grabCut( image, // input image
333  ret, // segmentation mask & result
334  cv::Rect(0,0,1,1), // rectangle containing foreground
335  cv::Mat(),cv::Mat(), // models
336  1, // number of iterations
337  cv::GC_INIT_WITH_MASK); // use rectangle
338 
339 
340  cv::compare(ret, cv::GC_PR_FGD , ret, cv::CMP_EQ);
341  return ret;
342  }

350  vector<Contour> generate_contours( const Mat1b& saliency_mask) {
351  using namespace cv;
352  vector<Contour> ret;
353 
354  vector<Vec4i> hierarchy;
355  findContours( saliency_mask, ret, hierarchy, CV_RETR_TREE, CV_CHAIN_APPROX_SIMPLE, Point(0, 0) );
356  for( int i=static_cast<int>(ret.size()-1); i >=0; --i) {
357  // just keep top-level contours of sufficient size
358  if( parent(hierarchy, i) != -1 || contourArea( ret[i]) < params.min_salient_region_size) {
359  ret.erase( ret.begin() + i);
360  }
361  }
362  return ret;
363  }
