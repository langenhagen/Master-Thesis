34  void eval_bayes( const program_options& params) {
35 
36  LOG(info) << "Loading class membership probabilities...";
37  Mat1r membership_probabilities;
38  exit_if_false( from_file( params.membership_probabilities_file, membership_probabilities), RETURN_CODE::IO_ERROR);
39 
40  const uint n_clusters = membership_probabilities.cols;
41  const uint n_features = membership_probabilities.rows;
42 
43  LOG(info) << "# features: " << n_features;
44  LOG(info) << "# clusters: " << n_clusters;
45 
46  LOG(info) << "Assigning membership mappings ...";
47  Vec1UInt membership_mappings = assign_membership( membership_probabilities);
48  vector<Mat1r> segmented_membership_probabilities( n_clusters);
49  for(uint r=0; r<n_features; ++r) {
50  segmented_membership_probabilities[membership_mappings[r]].push_back( membership_probabilities.row(r));
51  }
52  // === overall: calc P(C=c|X=x) ===
53 
54  LOG(info) << "Calculating likelihoods P(X=x|C=c)...";
55  // calc independent #classes-gaussian distributions ..for each class.
56  // likelihoods: P(X=x|C=c) = p(X_1=x_1|C=c)*p(X_2=x_2|C=c)*.. [class_idx][dimension]
57  vector<vector<RealGaussian>> likelihoods( n_clusters);
58  for( uint i=0; i<n_clusters; ++i) {
59 
60  const Mat1r& segment = segmented_membership_probabilities[i];
61 
62  // Check if segment matrix has full rank, which is important for checking
63  Mat1r singular_values;
64  cv::SVD::compute(segment, singular_values, cv::SVD::NO_UV);
65  if( std::count((app::real*)singular_values.datastart, (app::real*)singular_values.dataend, 0) > 0 ) {
66  LOG(warn) << "Segment matrix for class: " << i << " doesn't have full rank.";
67  }
68 
69  for( uint j=0; j<n_clusters; ++j) {
70  cv::Mat1d var, mean;
71  cv::calcCovarMatrix(segment.col(j), var, mean, CV_COVAR_NORMAL | CV_COVAR_ROWS); // splitted column-wise bc we only need the diagonals of the covar-matrix
72 
73  const real mean_value = static_cast<real>(mean(0,0));
74  const real var_value = static_cast<real>(var(0,0)/n_clusters);
75 
76  likelihoods[i].push_back( RealGaussian( mean_value, var_value));
77 
78  ANXIOUS_VALIDITY_CHECK( mean(0,0), "likelihoods - mean(0,0)");
79  ANXIOUS_VALIDITY_CHECK( var(0,0)/n_clusters, "likelihoods - var(0,0)/n_clusters");
80  }
81  }
82 
83  LOG(info) << "Calculating priors P(C=c)...";
84  Mat1r priors(1, n_clusters);
85  for(uint i=0; i<n_clusters; ++i) {
86  priors(0, i) = static_cast<real>(segmented_membership_probabilities[i].rows);
87  }
88  priors /= n_features;
89 
90  // do some numeric validity checks
91  for(uint i=0; i<n_clusters; ++i) {
92  std::stringstream ss;
93  ss << "priors(0, " << i << ")";
94  ANXIOUS_VALIDITY_CHECK( priors(0,i), ss.str());
95  }
96 
97 
98  LOG(info) << "Calculating (unnormalized) posteriors P(C=c|X=x)...";
99  Mat1r posteriors( n_features, n_clusters);
100  for(uint cluster=0; cluster<n_clusters; ++cluster) {
101 
102  const real prior = priors(0, cluster);
103 
104  for(uint feature=0; feature<n_features; ++feature) {
105 
106  real likelihood(1);
107  for( uint dim=0; dim<n_clusters; ++dim) {
108  real likelihood_1_dim = static_cast<real>( likelihoods[cluster][dim]( membership_probabilities(feature,dim)));
109  likelihood *= likelihood_1_dim;
110 
111  std::stringstream ss;
112  ss << "likelihood-1-dim: " << dim;
113  ANXIOUS_VALIDITY_CHECK( likelihood_1_dim, ss.str());
114  ss.str("");
115  ss << "likelihood - after " << dim << " multiplications";
116  ANXIOUS_VALIDITY_CHECK( likelihood, ss.str());
117  }
118 
119  posteriors( feature, cluster) = likelihood*prior;
120  ANXIOUS_VALIDITY_CHECK( likelihood*prior, "likelihood*prior");
121  }
122  }
123 
124  //TODO normalize TODO ????
125 
126  LOG(info) << "Writing posteriors to \"" << params.posteriors_file << "\"...";
127  to_file( params.posteriors_file, posteriors);
128 
129  }
