91  Mat1r normalized_fourier_descriptor( const Contour& contour, uint n_frequencies) const {
92  if (n_frequencies%2 != 0)
93  n_frequencies++;
94  Mat2r fd(1, static_cast<int>(contour.size()));
95  Mat2r normalized_fd( 1, n_frequencies);
96 
97  for( int c=0; c<fd.cols; ++c) {
98  fd(0, c)[0] = (float)contour[c].x;
99  fd(0, c)[1] = (float)contour[c].y;
100  }
101  cv::dft(fd, fd, cv::DFT_COMPLEX_OUTPUT); // no complex-conjugate symmetry, produce a full-size complex output array
102 
103  const real radius = sqrt( pow( fd(0, 1)[0], 2) +
104  pow( fd(0, 1)[1], 2));
105 
106  // take only low frequencies into account + normalize scale
107  for (uint i=0; i< n_frequencies>>2; ++i) {
108  // positive
109  normalized_fd(0, i)[0] = fd(0, i)[0] / radius;
110  normalized_fd(0, i)[1] = fd(0, i)[1] / radius;
111 
112  // negative --> lookat aia assignment2 page12 // lookat how opencv handles dft
113  normalized_fd(0, n_frequencies-1 - i)[0] = fd(0, fd.cols-1 - i)[0] / radius;
114  normalized_fd(0, n_frequencies-1 - i)[1] = fd(0, fd.cols-1 - i)[1] / radius;
115  }
116 
117  // translation invariance
118  normalized_fd(0,0)[0] = 0;
119  normalized_fd(0,0)[1] = 0;
120 
121  // split frequency spectrum into real and imaginary parts.
122  Mat1r planes[2]; // 0: Re(normalized_fd), 1: Im(normalized_fd)
123  split( normalized_fd, planes);
124 
125  Mat1r magnitude; // rotation invariant
126  Mat1r phase;
127  cartToPolar(planes[0], planes[1], magnitude, phase);
128 
129  return magnitude;
130  }
