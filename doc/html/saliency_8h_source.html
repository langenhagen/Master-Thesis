98  std::vector< float > uniqueness( const std::vector< SuperpixelStatistic >& stat ) {
99 
100  using namespace cv;
101 
102  const int N = stat.size();
103  std::vector< float > r( N );
104  const float sp = 0.5 / (settings_.sigma_p_ * settings_.sigma_p_);
105  for( int i=0; i<N; i++ ) {
106  float u = 0, norm = 1e-10;
107  Vec3f c = stat[i].mean_color_;
108  Vec2f p = stat[i].mean_position_;
109 
110  // Evaluate the score, for now without filtering
111  for( int j=0; j<N; j++ ) {
112  Vec3f dc = stat[j].mean_color_ - c;
113  Vec2f dp = stat[j].mean_position_ - p;
114 
115  float w = exp( - sp * dp.dot(dp) );
116  u += w*dc.dot(dc);
117  norm += w;
118  }
119  // Let's not normalize here, must have been a typo in the paper
120  // r[i] = u / norm;
121  r[i] = u;
122  }
123 
124  normVec(r );
125  return r;
126  }

128  std::vector< float > distribution( const std::vector< SuperpixelStatistic >& stat ) {
129 
130  using namespace cv;
131 
132  const int N = stat.size();
133  std::vector< float > r( N );
134  const float sc = 0.5 / (settings_.sigma_c_*settings_.sigma_c_);
135  for( int i=0; i<N; i++ ) {
136  float u = 0, norm = 1e-10;
137  Vec3f c = stat[i].mean_color_;
138  Vec2f p(0.f, 0.f);
139 
140  // Find the mean position
141  for( int j=0; j<N; j++ ) {
142  Vec3f dc = stat[j].mean_color_ - c;
143  float w = exp( - sc * dc.dot(dc) );
144  p += w*stat[j].mean_position_;
145  norm += w;
146  }
147  p *= 1.0 / norm;
148 
149  // Compute the variance
150  for( int j=0; j<N; j++ ) {
151  Vec3f dc = stat[j].mean_color_ - c;
152  Vec2f dp = stat[j].mean_position_ - p;
153  float w = exp( - sc * dc.dot(dc) );
154  u += w*dp.dot(dp);
155  }
156  r[i] = u / norm;
157  }
158  normVec( r );
159  return r;
160  }
161  std::vector< float > uniquenessFilter( const std::vector< SuperpixelStatistic >& stat ) {
162 
163  using namespace cv;
164 
165  const int N = stat.size();
166 
167  // Setup the data and features
168  std::vector< Vec2f > features( stat.size() );
169  Mat_<float> data( stat.size(), 5 );
170  for( int i=0; i<N; i++ ) {
171  features[i] = stat[i].mean_position_ / settings_.sigma_p_;
172  Vec3f c = stat[i].mean_color_;
173  data(i,0) = 1;
174  data(i,1) = c[0];
175  data(i,2) = c[1];
176  data(i,3) = c[2];
177  data(i,4) = c.dot(c);
178  }
179  // Filter
180  Filter filter( (const float*)features.data(), N, 2 );
181  filter.filter( data.ptr<float>(), data.ptr<float>(), 5 );
182 
183  // Compute the uniqueness
184  std::vector< float > r( N );
185  for( int i=0; i<N; i++ ) {
186  Vec3f c = stat[i].mean_color_;
187  float u = 0, norm = 1e-10;
188  Vec2f p = stat[i].mean_position_;
189 
190  r[i] = data(i,0)*c.dot(c) + data(i,4) - 2*( c[0]*data(i,1) + c[1]*data(i,2) + c[2]*data(i,3) );
191  }
192 
193  normVec( r );
194  return r;
195  }
196  std::vector< float > distributionFilter( const std::vector< SuperpixelStatistic >& stat ) {
197 
198  using namespace cv;
199 
200  const int N = stat.size();
201 
202  // Setup the data and features
203  std::vector< Vec3f > features( stat.size() );
204  Mat_<float> data( stat.size(), 4 );
205  for( int i=0; i<N; i++ ) {
206  features[i] = stat[i].mean_color_ / settings_.sigma_c_;
207  Vec2f p = stat[i].mean_position_;
208  data(i,0) = 1;
209  data(i,1) = p[0];
210  data(i,2) = p[1];
211  data(i,3) = p.dot(p);
212  }
213  // Filter
214  Filter filter( (const float*)features.data(), N, 3 );
215  filter.filter( data.ptr<float>(), data.ptr<float>(), 4 );
216 
217  // Compute the uniqueness
218  std::vector< float > r( N );
219  for( int i=0; i<N; i++ )
220  r[i] = data(i,3) / data(i,0) - ( data(i,1) * data(i,1) + data(i,2) * data(i,2) ) / ( data(i,0) * data(i,0) );
221 
222  normVec( r );
223  return r;
224  }

230  cv::Mat_< float > assignFilter( const cv::Mat_< cv::Vec3b >& im, const cv::Mat_< int >& seg, const std::vector< SuperpixelStatistic >& stat, const std::vector< float >& sal ) const {
231 
232  using namespace cv;
233 
234  std::vector< float > source_features( seg.size().area()*5 ), target_features( im.size().area()*5 );
235  Mat_< Vec2f > data( seg.size() );
236  // There is a type on the paper: alpha and beta are actually squared, or directly applied to the values
237  const float a = settings_.alpha_, b = settings_.beta_;
238 
239  const int D = 5;
240  // Create the source features
241  for( int j=0,k=0; j<seg.rows; j++ )
242  for( int i=0; i<seg.cols; i++, k++ ) {
243  int id = seg(j,i);
244  data(j,i) = Vec2f( sal[id], 1 );
245 
246  source_features[D*k+0] = a * i;
247  source_features[D*k+1] = a * j;
248  if (D == 5) {
249  source_features[D*k+2] = b * stat[id].mean_rgb_[0];
250  source_features[D*k+3] = b * stat[id].mean_rgb_[1];
251  source_features[D*k+4] = b * stat[id].mean_rgb_[2];
252  }
253  }
254  // Create the source features
255  for( int j=0,k=0; j<im.rows; j++ )
256  for( int i=0; i<im.cols; i++, k++ ) {
257  target_features[D*k+0] = a * i;
258  target_features[D*k+1] = a * j;
259  if (D == 5) {
260  target_features[D*k+2] = b * im(j,i)[0];
261  target_features[D*k+3] = b * im(j,i)[1];
262  target_features[D*k+4] = b * im(j,i)[2];
263  }
264  }
265 
266  // Do the filtering [Filtering using the target features twice works slightly better, as the method described in our paper]
267  if (settings_.use_spix_color_) {
268  Filter filter( source_features.data(), seg.cols*seg.rows, target_features.data(), im.cols*im.rows, D );
269  filter.filter( data.ptr<float>(), data.ptr<float>(), 2 );
270  }
271  else {
272  Filter filter( target_features.data(), im.cols*im.rows, D );
273  filter.filter( data.ptr<float>(), data.ptr<float>(), 2 );
274  }
275 
276  Mat_<float> r( im.size() );
277  for( int j=0; j<im.rows; j++ )
278  for( int i=0; i<im.cols; i++ )
279  r(j,i) = data(j,i)[0] / (data(j,i)[1] + 1e-10);
280  return r;
281  }

295  cv::Mat_<float>saliency( const cv::Mat_< cv::Vec3b >& im ) {
296  using namespace cv;
297 
298  // Convert the image to the lab space
299  cv::Mat_<cv::Vec3f> rgbim, labim;
300  im.convertTo( rgbim, CV_32F, 1.0/255. );
301  cv::cvtColor( rgbim, labim, CV_BGR2Lab );
302 
303  //std::cout << "\n\n" << "Doe abstract.";
304  //Mat_<int> segmentation = this->do_gSLIC(rgbim);
305 
306  Mat_<int> segmentation;
307  segmentation = superpixel_.geodesicSegmentation( labim );
308 
309  std::vector< SuperpixelStatistic > stat = superpixel_.stat( labim, im, segmentation );
310 
311  //std::cout << "\n" << "Doe uniqueness.";
312  // Compute the uniqueness
313  std::vector<float> unique( stat.size(), 1 );
314  if (settings_.uniqueness_) {
315  if (settings_.filter_uniqueness_)
316  unique = uniquenessFilter( stat );
317  else
318  unique = uniqueness( stat );
319  }
320 
321  //std::cout << "\n" << "Doe distrib.";
322  // Compute the distribution
323  std::vector<float> dist( stat.size(), 0 );
324  if (settings_.distribution_) {
325  if (settings_.filter_distribution_)
326  dist = distributionFilter( stat );
327  else
328  dist = distribution( stat );
329  }
330 
331  //std::cout << "\n" << "Combine unique & distrib.";
332  // Combine the two measures
333  std::vector<float> sp_saliency( stat.size() );
334  for( unsigned int i=0; i<stat.size(); ++i )
335  sp_saliency[i] = unique[i] * exp( - settings_.k_ * dist[i] );
336 
337  //std::cout << "\n" << "Doe upsamling.";
338  // Upsampling
339  Mat_<float> r;
340  if (settings_.upsample_)
341  r = assignFilter( im, segmentation, stat, sp_saliency );
342  else
343  r = assign( segmentation, sp_saliency );
344 
345  //std::cout << "\n" << "Rescale saliency.";
346  // Rescale the saliency to [0..1]
347  double mn, mx;
348  minMaxLoc( r, &mn, & mx );
349  r = (r - mn) / (mx - mn);
350 
351  //todo std::cout << "\n" << "increase sal level.";
353  double m_sal = settings_.min_saliency_ * r.size().area();
354  //for( float sm = sum( r )[0]; sm < m_sal; sm = sum( r )[0] )
355  // r = min( r*m_sal/sm, 1.0f );
356 
357  //std::cout << "\n" << "Finished.\n\n";
358  return r;
359  }

364  // Normalize a vector of floats to the range [0..1]
365  void normVec( std::vector<float>& r ) {
366  const int N = r.size();
367  float mn = r[0], mx = r[0];
368  for( int i=1; i<N; i++ ) {
369  if (mn > r[i])
370  mn = r[i];
371  if (mx < r[i])
372  mx = r[i];
373  }
374  for( int i=0; i<N; i++ )
375  r[i] = (r[i] - mn) / (mx - mn);
376  }
