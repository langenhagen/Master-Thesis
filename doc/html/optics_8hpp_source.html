102  void expand_cluster_order( DataVector& db, DataPoint* p, const real eps, const unsigned int min_pts, DataVector& o_ordered_vector) {
103  assert( eps >= 0 && "eps must not be negative");
104  assert( min_pts > 0 && "min_pts must be greater than 0");
105 
106  DataVector N_eps = get_neighbors( p, eps, db);
107  p->reachability_distance( OPTICS::UNDEFINED);
108  const real core_dist_p = squared_core_distance( p, min_pts, N_eps);
109  p->processed( true);
110  o_ordered_vector.push_back( p);
111 
112  if( core_dist_p == OPTICS::UNDEFINED)
113  return;
114 
115  DataSet seeds;
116  update_seeds( N_eps, p, core_dist_p, seeds);
117 
118  while( !seeds.empty()) {
119  DataPoint* q = *seeds.begin();
120  seeds.erase( seeds.begin()); // remove first element from seeds
121 
122  DataVector N_q = get_neighbors( q, eps, db);
123  const real core_dist_q = squared_core_distance( q, min_pts, N_q);
124  q->processed( true);
125  o_ordered_vector.push_back( q);
126  if( core_dist_q != OPTICS::UNDEFINED) {
127  // *** q is a core-object ***
128  update_seeds( N_q, q, core_dist_q, seeds);
129  }
130  }
131  }

229  void update_seeds( const DataVector& N_eps, const DataPoint* center_object, const real c_dist, DataSet& o_seeds) {
230  assert( c_dist != OPTICS::UNDEFINED && "the core distance must be set <> UNDEFINED when entering update_seeds");
231 
232  for( DataVector::const_iterator it=N_eps.begin(); it!=N_eps.end(); ++it) {
233  DataPoint* o = *it;
234 
235  if( o->is_processed())
236  continue;
237 
238  const real new_r_dist = std::max( c_dist, squared_distance( center_object, o));
239  // *** new_r_dist != UNDEFINED ***
240 
241  if( o->reachability_distance() == OPTICS::UNDEFINED) {
242  // *** o not in seeds ***
243  o->reachability_distance( new_r_dist);
244  o_seeds.insert( o);
245 
246  } else if( new_r_dist < o->reachability_distance()) {
247  // *** o already in seeds & can be improved ***
248  o_seeds.erase( o);
249  o->reachability_distance( new_r_dist);
250  o_seeds.insert( o);
251  }
252  }
253  }

285  real squared_core_distance( const DataPoint* p, const unsigned int min_pts, DataVector& N_eps) {
286  assert( min_pts > 0 && "min_pts must be greater than 0");
287  real ret( OPTICS::UNDEFINED);
288 
289  if( N_eps.size() > min_pts) {
290  std::nth_element( N_eps.begin(),
291  N_eps.begin()+min_pts,
292  N_eps.end(),
293  [p]( const DataPoint* a, const DataPoint* b){ return squared_distance( p, a) < squared_distance( p, b); } );
294 
295  ret = squared_distance( p, N_eps[min_pts]);
296  }
297  return ret;
298  }

305  real squared_distance( const DataPoint* a, const DataPoint* b) {
306  const std::vector<real>& a_data = a->data();
307  const std::vector<real>& b_data = b->data();
308  const unsigned int vec_size = static_cast<unsigned int>(a_data.size());
309  assert( vec_size == b_data.size() && "Data-vectors of both DataPoints must have same dimensionality");
310  real ret(0);
311 
312  for( unsigned int i=0; i<vec_size; ++i) {
313  ret += std::pow( a_data[i]-b_data[i], 2);
314  }
315  //return std::sqrt( ret);
316  return ret;

352  std::vector<DataVector> extract_clusters( const DataVector& result, const std::vector<unsigned int>& cluster_borders, real outlier_threshold) {
353  std::vector<DataVector> ret;
354  ret.push_back( DataVector()); // outlier container
355 
356  if( outlier_threshold <= 0)
357  outlier_threshold = std::numeric_limits<real>::max();
358 
359 
360  for( unsigned int i=0; i<=cluster_borders.size(); ++i) {
361 
362  const unsigned int lower_idx = i == 0 ? 0 : cluster_borders[i-1];
363  const unsigned int upper_idx = i == cluster_borders.size() ? static_cast<unsigned int>(result.size()) : cluster_borders[i];
364 
365  DataVector cluster_i;
366 
367  for( unsigned int j=lower_idx; j<upper_idx; ++j) {
368  DataPoint* p = result[j];
369 
370  if( p->reachability_distance() > outlier_threshold) {
371  ret[0].push_back( p);
372  } else {
373  cluster_i.push_back( p);
374  }
375  }
376  ret.push_back( cluster_i);
377  }
378  return ret;
379  }
