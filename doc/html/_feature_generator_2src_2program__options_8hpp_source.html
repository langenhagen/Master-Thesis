141  bool adjust_program_options( program_options& p) {
142  bool ret(true);
143 
144  if( p.blur_kernel_size_uint % 2 == 0) {
145  LOG(warn) << "Given blur_kernel_size must be an odd number! Adjusting to " << ++p.blur_kernel_size_uint;

148 
149  p.sdd.type = detector_type_from_string( p.sdd.type_string);
150  if( p.sdd.type == detector_type::ERROR_TYPE)
151  ret = false;
152  p.sdd.tweak_vector = from_string<real,vector>( p.sdd.tweak_vector_string);
153 
154  p.fed.type = extractor_from_string( p.fed.type_string);
155  if( p.fed.type == extractor_type::ERROR_TYPE)
156  ret = false;
157  p.fed.tweak_vector = from_string<real,vector>( p.fed.tweak_vector_string);
158 
159  if(ret==false) {
160  LOG( error) << "Not correctable error in program options!";
161  }
162  return ret;
163  }

173  bool init_program_options( const int argc, const char* argv[], program_options& p) {
174  using namespace boost::program_options;
175  using namespace std;
176  bool ret(true);
177 
178  // declare the options ************************************************
179 
180  // 1. group of options that will be allowed only on command line
181  options_description cmd_desc("Command line options options");
182  cmd_desc.add_options()
183  ("help", "produce help message")
184  ("config,c", value<string>(&p.config_file)->default_value("config.cfg"), "name of the config file")
185  ;
186  // 2. group description on the config file
187  options_description config_desc("Configuration file parameters");
188  config_desc.add_options()
189  ("log_file", value<string>(&p.log_file)->default_value("log.log"), "name of the log file")
190  ("directories_file", value<string>(&p.directories_file)->default_value("directories.txt"), "name of the file that stores the image directories")
191  ("include_subdirs", value<bool>(&p.include_subdirs)->default_value(1), "whether or not to include subdirectories in looking through the image databases")
192  ("output_directory", value<string>(&p.output_directory)->default_value("out"), "the output directory for eventual output-files")
193  ("features_file", value<string>(&p.features_file)->default_value("features.desc"), "a file that stores the salient object feature vectors")
194  ("processed_images_file", value<string>(&p.processed_images_file)->default_value("processed_files.txt"), "a file that stores the paths of the already processed images")
195  ("garbage_file", value<string>(&p.garbage_file)->default_value("files_with_no_salient_regions.txt"), "a file that stores the paths images in which no salient region was found")
196  ("delete_old_features", value<bool>(&p.delete_old_features)->default_value(false), "whether or not to delete/reuse the feature vectors generated in a previous run")
197  ("blur_kernel_size", value<uint>(&p.blur_kernel_size_uint), "size of the blur filter to smooth the saliency masks in each dimension")
198  ("threshold", value<real>(&p.threshold), "the threshold of deciding when a pixel of the saliency map is part of a salient object. Should be within ]0;1[")
199  ("min_salient_region_size", value<real>(&p.min_salient_region_size)->default_value(0), "the minimum size of a salient region in pixels")
200  ("use_grabcut", value<bool>(&p.use_grabcut)->default_value(false), "Whether or not to use GrabCut for creation of saliency masks")
201  ("grabcut_foreground_probability", value<real>(&p.grabcut_foreground_probability)->default_value(static_cast<real>(0.2)), "The probability of a pixel to be long to the foreground")
202  ("save_saliency_maps", value<bool>(&p.save_saliency_maps)->default_value(0), "whether or not to save saliency maps to disk")
203  ("saliency_maps_file", value<string>(&p.saliency_maps_file)->default_value("saliency_maps.txt"), "stores the paths to eventually created saliency maps")
204  ("save_saliency_masks", value<bool>(&p.save_saliency_masks)->default_value(false), "whether or not to save the saliency masks to disk")
205  ("saliency_masks_file", value<string>(&p.saliency_masks_file)->default_value("saliency_masks.txt"), "stores the paths to eventually created saliency masks")
206  ("symlink_garbage_files", value<bool>(&p.symlink_garbage_files)->default_value(false), "whether or not to symlink the files without salient regions")
207  ("detector_type", value<string>(&p.sdd.type_string), detector_types_string().c_str())
208  ("detector_tweak_vector", value<string>(&p.sdd.tweak_vector_string)->default_value(""), "real-numeric tweaks for the saliency detector separated by spaces \" \".")
209  ("extractor_type", value<string>(&p.fed.type_string), extractor_types_string().c_str())
210  ("extractor_tweak_vector", value<string>(&p.fed.tweak_vector_string)->default_value(""), "real-numeric tweaks for the feature extractor separated by spaces \" \".")
211  ;
212  // END option declarations ********************************************
213 
214  // *** parse the command line and the config files ***
215  variables_map vm;
216 
217  // 1. parse command line ******************************************
218  store(parse_command_line(argc, argv, cmd_desc), vm);
219  boost::program_options::notify(vm);
220 
221  if( vm.count("help")) {
222  // special 'help' case
223  print_help_message( 2, &cmd_desc, &config_desc);
224  return false;
225  }
226  // 2. parse config file *******************************************
227  ifstream in_file(p.config_file.c_str());
228  if (!in_file) {
229  std::cerr << "Can not open config file: " << p.config_file << "\n"
230  "Did you create a config file or do you want to specify another config file?\n";
231  ret = false;
232  } else {
233  // *** file access works ***
234  parsed_options po (parse_config_file(in_file, config_desc, true));
235  store( po, vm);
236  boost::program_options::notify(vm);
237  }
238  in_file.close();
239  return ret;
240  }
