777  template<typename R>
778  std::vector<R> circular_cross_correlation_series( const std::vector<R>& a, const std::vector<R>& b) {
779  assert( a.size() == b.size() && "a and b must have same length");
780  std::vector<R> ret;
781  unsigned int size = static_cast<unsigned int>(a.size()/2) + (a.size()%2 == 0 ? 0 : 1);
782  ret.reserve( size);
783 
784  register R mean_a = 0;
785  register R mean_b = 0;
786  R variance_a = 0;
787  R variance_b = 0;
788  register R denom;
789 
790  // calculate means
791  for( uint i=0; i<a.size(); ++i) {
792  mean_a += a[i];
793  mean_b += b[i];
794  }
795  mean_a /= a.size();
796  mean_b /= a.size();
797  // calculate variances
798  for( uint i=0; i<a.size(); ++i) {
799  variance_a += (a[i] - mean_a) * (a[i] - mean_a);
800  variance_b += (b[i] - mean_b) * (b[i] - mean_b);
801  }
802  denom = sqrt( variance_a * variance_b);
803 
804  // calculate correlation series
805  if( denom == 0) {
806  for( unsigned int delay=0; delay<size; ++delay) {
807  ret.push_back(1);
808  }
809  } else {
810  for( unsigned int delay=0; delay<size; ++delay) {
811  R cross_correlation = 0;
812  for( uint idx_a=0; idx_a<a.size(); ++idx_a) {
813  int idx_b = (idx_a+delay) % a.size();
814  cross_correlation += (a[idx_a] - mean_a) * (b[idx_b] - mean_b);
815  }
816  ret.push_back( cross_correlation / denom);
817  }
818  }
819  return ret;
820  }
