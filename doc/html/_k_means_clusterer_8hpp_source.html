54  virtual Mat1r do_cluster( const Mat1r& features) const {
55  check_and_resolve_input_errors();
56  Mat1r ret;
57  int n_clusters;
58  Mat1r cluster_means( static_cast<int>(description.tweak_vector[0]), features.cols);
59 
60  cvflann::KMeansIndexParams kmeans_index_params( 32, -1, cvflann::CENTERS_KMEANSPP, 0.2f);
61  n_clusters = cv::flann::hierarchicalClustering<cv::flann::L2<real>>( features, cluster_means, kmeans_index_params);
62  ret = Mat1r( features.rows, n_clusters, real(0));
63 
64  for( int r=0; r<features.rows; ++r) {
65  // assign the probility of each feature to one cluster. k means is hard assignment.
66  const cv::Mat_<real> &feat = features.row(r);
67 
68  // get nearest cluster
69  real min_dist = real( norm( feat, cluster_means.row(0), cv::NORM_L2));
70  int nearest_cluster(0);
71  for( int c=1; c<n_clusters; ++c) {
72  const real dist = real( norm( feat, cluster_means.row(c), cv::NORM_L2));
73  if( dist < min_dist) {
74  min_dist = dist;
75  nearest_cluster = c;
76  }
77  }
78 
79  // assign probability 1 to nearest cluster
80  ret(r, nearest_cluster) = 1;
81  }
82 
83  return ret;
84  }
