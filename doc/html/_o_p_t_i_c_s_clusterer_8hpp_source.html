58  virtual Mat1r do_cluster( const Mat1r& features) const {
59  typedef OPTICS::LabelledDataPoint<int> OPTICSPoint;
60  const uint n_features = features.rows;
61  check_and_resolve_input_errors( n_features);
62  Mat1r ret;
63 
64  const Vec1r& tweak = this->description.tweak_vector;
65 
66  const optics_mode mode = tweak[0] == N_CLUSTERS ? N_CLUSTERS : PERSISTENCE; // not schoen but selten
67  const OPTICS::real eps = tweak[1];
68  const uint min_pts = static_cast<uint>(tweak[2]);
69  const uint n_clusters = static_cast<uint>(tweak[3]);
70  const OPTICS::real persistence = tweak[3];
71  const OPTICS::real outlier_threshold = tweak[4];
72 
73  // assemble data
74  OPTICS::DataVector db;
75  for( int r=0; r<features.rows; ++r) {
76  OPTICSPoint* p = new OPTICSPoint( r);
77  vector<OPTICS::real>& data = p->data();
78  for( int c=0; c<features.cols; ++c)
79  data.push_back( features(r,c));
80  db.push_back(p);
81  }
82 
83  // run optics
84  uint n_processed = 0;
85  OPTICS::DataVector result =
86  OPTICS::optics( db,
87  eps,
88  min_pts,
89  [&n_processed, &n_features](const OPTICS::DataPoint* p){
90  n_processed++;
91  if( n_processed % 100 == 0) {
92  const real percent = static_cast<int>( 100.0 * n_processed / n_features * 100 + 0.5) / 100.0f;
93  LOG(info) << "OPTICSClusterer: " << percent << "% (" << n_processed << '/' << n_features << ") done.";
94  }
95  });
96 
97  // extract reachability distances
98  vector<OPTICS::real> reachabilities;
99  Vec1i ordered_indices;
100  std::for_each( result.begin(),
101  result.end(),
102  [&reachabilities, &ordered_indices]( const OPTICS::DataPoint* p) {
103  const OPTICSPoint* lp = dynamic_cast<const OPTICSPoint*>(p);
104  reachabilities.push_back( p->reachability_distance());
105  ordered_indices.push_back( lp->label());
106  });
107 
108 
109  const string reachability_distances_fname = "optics_reachability_distances.txt";
110  const string ordered_indices_fname = "optics_ordered_feature_indices.txt";
111  LOG(info) << "OPTICSClusterer: Writing reachability histogram to \"" << reachability_distances_fname << "\"...";
112  to_file(reachability_distances_fname, reachabilities);
113  LOG(info) << "OPTICSClusterer: Writing ordered indices to \"" << ordered_indices_fname << "\"...";
114  to_file(ordered_indices_fname, ordered_indices);
115 
116 
117  // find clusters
118  Vec1UInt cluster_borders;
119  if( mode == N_CLUSTERS) {
120  cluster_borders = find_k_histogram_peaks( reachabilities, n_clusters);
121  } else if( mode == PERSISTENCE) {
122  cluster_borders = find_histogram_peaks( reachabilities, persistence);
123  } else {
124  //should never happen
125  LOG(error) << "OPTICSClusterer: Not able to handle the mode \"" << mode << "\".";
126  }
127 
128  std::sort( cluster_borders.begin(), cluster_borders.end());
129  vector<OPTICS::DataVector> clusters = OPTICS::extract_clusters( result, cluster_borders, outlier_threshold);
130 
131 
132  // set 1 on assigned cluster position
133  ret = Mat1r( n_features, static_cast<int>(clusters.size()), real(0));
134  for( uint i=0; i<clusters.size(); ++i) {
135  const OPTICS::DataVector& cluster_i = clusters[i];
136  for( uint j=0; j<cluster_i.size(); ++j) {
137  const OPTICSPoint* p = dynamic_cast<OPTICSPoint*>(cluster_i[j]);
138  auto tt = p->label();
139  ret( p->label(), i) = 1;
140  }
141  }
142 
143  // clean up
144  for( auto it=db.begin(); it!=db.end(); ++it) {
145  delete *it;
146  }
147 
148 
149  return ret;

159  void check_and_resolve_input_errors( const uint n_features) const {
160  Vec1r& tweak = this->description.tweak_vector;
161 
162  if( tweak.size() < 5 ||
163  tweak[0] != N_CLUSTERS && tweak[0] != PERSISTENCE || // mode (N_CLUSTERS or PERSISTENCE)
164  tweak[1] <= 0 || // epsilon (el. R+)
165  tweak[2] <= 0 || // min_pts (el. N+)
166  tweak[3] <= 0 || // n_clusters (el N+) or persistence (el. R+)
167  tweak[4] <= 0 // outlier_threshold (el. R+)
168  ) {
169 
170  const OPTICS::real max = std::numeric_limits<OPTICS::real>::max();
171  const real min_pts_default_percentage = 1;
172 
173  LOG(warn) << "OPTICSClusterer: Tweak vector must contain 5 parameters:\n"
174  "0: 1 number specifying the behaviour: 0: find n clusters or 1: find all clusters with high persistence at peaks\n"
175  "1: the epsilon optics parameter, if zero or negative, epsilon will be set to " << max << "\n"
176  "2: the min_pts optics parameter, a positive integer; will otherwise be set to " << min_pts_default_percentage << "% of the size of the input data set\n"
177  "3: the number of clusters, if parameter 0 is set to '0' or the persistence value if parameter 0 is set to '1'\n"
178  "4: the outlier threshold: must be positive, will be set to " << max << " otherwise";
179 
180  tweak.resize(5, -1); // if too few parameters where given
181 
182  // mode
183  if( tweak[0] != N_CLUSTERS && tweak[0] != PERSISTENCE) {
184  tweak[0] = N_CLUSTERS;
185  LOG(notify) << "Setting OPTICS mode to N_CLUSTERS aka " << tweak[0] << ".";
186  }
187  // epsilon
188  if( tweak[1] <=0) {
189  tweak[1] = max;
190  LOG(notify) << "Setting epsilon to " << tweak[1] << ".";
191  }
192  // min_pts
193  if( tweak[2] <=0) {
194  tweak[2] = static_cast<real>(n_features) / 100 * min_pts_default_percentage;
195  LOG(notify) << "Setting min_pts to " << tweak[2] << ".";
196  }
197  // n_clusters / persistence
198  if( tweak[3] <=0) {
199  if( tweak[0] == N_CLUSTERS) {
200  tweak[3] = 1;
201  LOG(notify) << "Setting n_clusters to " << tweak[3] << ".";
202  } else {
203  tweak[3] = 0;
204  LOG(notify) << "Setting persistence value to " << tweak[3] << ".";
205  }
206  }
207  // outlier_threshold
208  if( tweak[4] <=0) {
209  tweak[4] = max;
210  LOG(notify) << "Setting outlier threshold value to " << tweak[4] << ".";
211  }
212 
213  } // END IF
214  }
